// 处理项目类型变更

function handleProjectChange() {

const type = document.getElementById('projectType').value;

// 隐藏所有特殊输入

document.getElementById('baseAmountGroup').style.display = 'block';

document.getElementById('auditInputs').classList.remove('show');

document.getElementById('steelInputs').classList.remove('show');

document.getElementById('manDayInputs').classList.remove('show');

document.getElementById('disputeInputs').classList.remove('show');

// 更新标签

const config = rateTables[type];

if (config) {

document.getElementById('baseUnit').textContent = type === '8' ? '吨' : '万元';

document.getElementById('tierInfo').textContent = generateTierInfo(config);

}

// 特殊处理

if (type === '5-1' || type === '5-2') {

document.getElementById('auditInputs').classList.add('show');

} else if (type === '8') {

document.getElementById('baseAmountGroup').style.display = 'none';

document.getElementById('steelInputs').classList.add('show');

} else if (type === '9') {

document.getElementById('baseAmountGroup').style.display = 'none';

document.getElementById('manDayInputs').classList.add('show');

} else if (type.startsWith('7')) {

document.getElementById('disputeInputs').classList.add('show');

if (type === '7-2') {

document.getElementById('bothParties').checked = true;

updateDisputeRate();

}

}

// 隐藏结果

document.getElementById('resultCard').classList.remove('show');

}

// 差额定率累进计算

function calculateProgressive(amount, rates) {

let fee = 0;

let remaining = amount;

let prevLimit = 0;

for (let tier of rates) {

const tierAmount = Math.min(remaining, tier.limit - prevLimit);

if (tierAmount <= 0) break;

fee += tierAmount * tier.rate;

remaining -= tierAmount;

prevLimit = tier.limit;

if (remaining <= 0) break;

}

return fee;

}

<p>中级资格专业人员：${intermediate}小时 × 100元 = ${intermediateFee.toFixed(2)}元</p>

<p>初级资格专业人员：${junior}小时 × 60元 = ${juniorFee.toFixed(2)}元</p>

<p><strong>合计：${fee.toFixed(2)}元</strong></p>

`;

}

}

<p>收费标准：12元/吨</p>

}

return;

}

// 结算审核特殊处理

if (type === '5-1' || type === '5-2') {

const auditAmount = parseFloat(document.getElementById('auditAmount').value) || 0;

const reduceAmount = parseFloat(document.getElementById('reduceAmount').value) || 0;

const increaseAmount = parseFloat(document.getElementById('increaseAmount').value) || 0;

if (auditAmount <= 0) {

alert('请输入送审结算价');

return;

}

// 基本收费

const config = rateTables['5-1'];

const basicFee = calculateProgressive(auditAmount, config.rates);

// 效益收费

const benefitBase = reduceAmount + increaseAmount;

const benefitFee = benefitBase * 0.05; // 5%

if (type === '5-1') {

fee = basicFee;

detail = `

<p><strong>基本收费计算（差额定率累进）：</strong></p>

<p>送审结算价：${auditAmount}万元</p>

<p>计算过程：${generateProgressiveDetail(auditAmount, config.rates)}</p>

<p>基本收费：${basicFee.toFixed(2)}万元 = ${(basicFee * 10000).toFixed(2)}元</p>

`;

baseAmount = auditAmount;

} else {

fee = basicFee + benefitFee;

detail = `

<p><strong>总收费 = 基本收费 + 效益收费</strong></p>

<p><strong>基本收费：</strong></p>

<p>送审结算价：${auditAmount}万元</p>

<p>计算过程：${generateProgressiveDetail(auditAmount, config.rates)}</p>

<p>基本收费：${basicFee.toFixed(2)}万元</p>

<p><strong>效益收费：</strong></p>

<p>核减额：${reduceAmount}万元 | 核增额：${increaseAmount}万元</p>

<p>效益收费基数：${benefitBase}万元</p>

<p>费率：5%</p>

<p>效益收费：${benefitFee.toFixed(2)}万元</p>

<p><strong>合计：${fee.toFixed(2)}万元 = ${(fee * 10000).toFixed(2)}元</strong></p>

`;

baseAmount = auditAmount;

}

// 转换元并检查最低收费

fee = fee * 10000;

if (fee < 2000 && fee > 0) {

detail += `<p style="color:#f57c00;">注：按最低收费标准调整为2000元</p>`;

fee = 2000;

}

showResult(fee, detail);

return;

}

// 普通差额定率累进计算

baseAmount = parseFloat(document.getElementById('baseAmount').value) || 0;

if (baseAmount <= 0) {

alert('请输入有效的收费基数');

return;

}

const config = rateTables[type];

if (!config) {

alert('请选择有效的咨询项目类型');

return;

}

// 纠纷鉴证特殊处理（双方有造价）

if (type === '7-2') {

const threshold = 1000;

if (baseAmount <= threshold) {

fee = baseAmount * 0.05;

detail = `

<p><strong>计费方式：</strong>双方均有造价</p>

<p>争议差额：${baseAmount}万元（1000万元以内）</p>

<p>费率：5%</p>

<p>计算：${baseAmount} × 5% = ${fee.toFixed(2)}万元</p>

`;

} else {

const lowPart = threshold * 0.05;

const highPart = (baseAmount - threshold) * 0.04;

fee = lowPart + highPart;

detail = `

<p><strong>计费方式：</strong>双方均有造价</p>

<p>争议差额：${baseAmount}万元（超过1000万元）</p>

<p>1000万元以内部分：1000 × 5% = ${lowPart.toFixed(2)}万元</p>

<p>1000万元以上部分：${(baseAmount - 1000).toFixed(2)} × 4% = ${highPart.toFixed(2)}万元</p>

<p>合计：${fee.toFixed(2)}万元</p>

`;

}

} else {

// 标准差额定率累进

fee = calculateProgressive(baseAmount, config.rates);

<p><strong>收费基数：</strong>${baseAmount}万元（${config.baseName}）</p>

<p><strong>差额定率累进计算：</strong></p>

<p>${generateProgressiveDetail(baseAmount, config.rates)}</p>

<p>合计：${fee.toFixed(4)}万元</p>

`;

}

}

// 转换元并检查最低收费

fee = fee * 10000;

if (fee < 2000 && fee > 0) {

detail += `<p style="color:#f57c00;">注：按最低收费标准调整为2000元</p>`;

fee = 2000;

}

showResult(fee, detail);

}

// 生成累进计算明细

function generateProgressiveDetail(amount, rates) {

let detail = [];

let remaining = amount;

let prevLimit = 0;

for (let tier of rates) {

const tierAmount = Math.min(remaining, tier.limit - prevLimit);

if (tierAmount <= 0) break;

const tierFee = tierAmount * tier.rate;

const rateDisplay = (tier.rate * 1000).toFixed(1);

if (tier.limit === Infinity) {

detail.push(`${tierAmount.toFixed(2)}万元 × ${rateDisplay}‰ = ${tierFee.toFixed(4)}万元`);

} else {

detail.push(`${tierAmount.toFixed(2)}万元 × ${rateDisplay}‰ = ${tierFee.toFixed(4)}万元`);

}

remaining -= tierAmount;

prevLimit = tier.limit;

if (remaining <= 0) break;

}

return detail.join('<br>');

}

// 显示结果

minimumFractionDigits: 2,

document.getElementById('resultDetail').innerHTML = detailHtml;

document.getElementById('resultCard').classList.add('show');

// 滚动到结果区域

document.getElementById('resultCard').scrollIntoView({ behavior: 'smooth', block: 'nearest' });

}
